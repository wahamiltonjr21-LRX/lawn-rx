import React, { useMemo, useState } from "react";
import { Camera, Upload, Leaf, Sparkles, CheckCircle2, AlertTriangle, Droplets, Sun, ClipboardList } from "lucide-react";
import { Button } from "@/components/ui/button";
import { Card, CardContent } from "@/components/ui/card";

export default function LawnIQPrototype() {
  const [photo, setPhoto] = useState(null);
  const [issueColor, setIssueColor] = useState("Yellow grass");
  const [description, setDescription] = useState("");
  const [grassType, setGrassType] = useState("Unknown");
  const [analyzed, setAnalyzed] = useState(false);

  const diagnosis = useMemo(() => {
    const text = `${issueColor} ${description}`.toLowerCase();

    if (text.includes("brown") || text.includes("patch")) {
      return {
        title: "Possible Brown Patch / Water Stress",
        score: 64,
        severity: "Medium",
        confidence: "78%",
        summary: "Your lawn may be showing signs of fungal activity or uneven watering. Circular brown areas often appear when humidity is high or watering happens late in the day.",
        steps: [
          "Water early in the morning only.",
          "Avoid watering at night.",
          "Rake out dead grass gently to improve airflow.",
          "Apply a lawn fungicide if spots continue spreading.",
        ],
      };
    }

    if (text.includes("weed")) {
      return {
        title: "Weed Pressure Detected",
        score: 70,
        severity: "Medium",
        confidence: "74%",
        summary: "The lawn may be thinning, which allows weeds to compete with turf. Improving grass density is the best long-term fix.",
        steps: [
          "Identify whether weeds are broadleaf or grassy weeds.",
          "Use a selective herbicide that matches the weed type.",
          "Mow at the correct height for your grass type.",
          "Overseed thin areas after weed control.",
        ],
      };
    }

    if (text.includes("dry") || text.includes("crispy")) {
      return {
        title: "Likely Drought or Heat Stress",
        score: 58,
        severity: "High",
        confidence: "81%",
        summary: "Dry, brittle grass usually points to insufficient water, compacted soil, or heat stress.",
        steps: [
          "Water deeply 2–3 times per week instead of daily shallow watering.",
          "Check soil moisture 3 inches deep.",
          "Aerate compacted areas if water runs off.",
          "Avoid fertilizing during extreme heat stress.",
        ],
      };
    }

    return {
      title: "Likely Nitrogen Deficiency",
      score: 72,
      severity: "Low to Medium",
      confidence: "83%",
      summary: "Yellowing grass often suggests low nitrogen, especially when the lawn looks pale, thin, or slow-growing.",
      steps: [
        "Apply a slow-release nitrogen fertilizer.",
        "Water after fertilizer application.",
        "Retest lawn color after 10–14 days.",
        "Avoid over-fertilizing, which can burn the grass.",
      ],
    };
  }, [issueColor, description]);

  function handlePhotoUpload(event) {
    const file = event.target.files?.[0];
    if (!file) return;
    const url = URL.createObjectURL(file);
    setPhoto(url);
    setAnalyzed(false);
  }

  return (
    <div className="min-h-screen bg-gradient-to-br from-emerald-50 via-white to-lime-50 p-4 text-slate-900">
      <div className="mx-auto max-w-md space-y-4">
        <header className="rounded-3xl bg-white p-5 shadow-sm border border-emerald-100">
          <div className="flex items-center gap-3">
            <div className="rounded-2xl bg-emerald-100 p-3">
              <Leaf className="h-7 w-7 text-emerald-700" />
            </div>
            <div>
              <h1 className="text-2xl font-bold">LawnIQ AI</h1>
              <p className="text-sm text-slate-600">Scan. Diagnose. Fix your lawn.</p>
            </div>
          </div>
        </header>

        <Card className="rounded-3xl shadow-sm border-emerald-100">
          <CardContent className="p-5 space-y-4">
            <div>
              <h2 className="text-lg font-semibold flex items-center gap-2">
                <Camera className="h-5 w-5" /> Upload Lawn Photo
              </h2>
              <p className="text-sm text-slate-600">Take a clear photo in daylight or upload from your gallery.</p>
            </div>

            <label className="block cursor-pointer rounded-3xl border-2 border-dashed border-emerald-200 bg-emerald-50 p-5 text-center hover:bg-emerald-100 transition">
              {photo ? (
                <img src={photo} alt="Uploaded lawn" className="h-48 w-full rounded-2xl object-cover" />
              ) : (
                <div className="space-y-2">
                  <Upload className="mx-auto h-10 w-10 text-emerald-700" />
                  <p className="font-medium">Tap to upload photo</p>
                  <p className="text-xs text-slate-500">Camera or gallery works on mobile</p>
                </div>
              )}
              <input type="file" accept="image/*" capture="environment" className="hidden" onChange={handlePhotoUpload} />
            </label>
          </CardContent>
        </Card>

        <Card className="rounded-3xl shadow-sm border-emerald-100">
          <CardContent className="p-5 space-y-4">
            <h2 className="text-lg font-semibold flex items-center gap-2">
              <ClipboardList className="h-5 w-5" /> Lawn Details
            </h2>

            <div className="space-y-2">
              <label className="text-sm font-medium">What does your lawn look like?</label>
              <select
                value={issueColor}
                onChange={(e) => setIssueColor(e.target.value)}
                className="w-full rounded-2xl border border-slate-200 bg-white p-3 text-sm"
              >
                <option>Yellow grass</option>
                <option>Brown patches</option>
                <option>Dry or crispy</option>
                <option>Weeds</option>
                <option>Thin / bare spots</option>
              </select>
            </div>

            <div className="space-y-2">
              <label className="text-sm font-medium">Grass type</label>
              <select
                value={grassType}
                onChange={(e) => setGrassType(e.target.value)}
                className="w-full rounded-2xl border border-slate-200 bg-white p-3 text-sm"
              >
                <option>Unknown</option>
                <option>Tall Fescue</option>
                <option>Bermuda</option>
                <option>Zoysia</option>
                <option>Kentucky Bluegrass</option>
                <option>St. Augustine</option>
              </select>
            </div>

            <div className="space-y-2">
              <label className="text-sm font-medium">Describe the problem</label>
              <textarea
                value={description}
                onChange={(e) => setDescription(e.target.value)}
                placeholder="Example: yellow spots near the driveway, grass looks weak..."
                className="min-h-24 w-full rounded-2xl border border-slate-200 bg-white p-3 text-sm"
              />
            </div>

            <Button
              onClick={() => setAnalyzed(true)}
              className="w-full rounded-2xl bg-emerald-700 py-6 text-base hover:bg-emerald-800"
            >
              <Sparkles className="mr-2 h-5 w-5" /> Analyze Lawn
            </Button>
          </CardContent>
        </Card>

        {analyzed && (
          <Card className="rounded-3xl shadow-md border-emerald-200">
            <CardContent className="p-5 space-y-5">
              <div className="flex items-start justify-between gap-3">
                <div>
                  <p className="text-sm text-slate-500">AI Diagnosis</p>
                  <h2 className="text-xl font-bold">{diagnosis.title}</h2>
                </div>
                <div className="rounded-2xl bg-amber-100 px-3 py-2 text-sm font-semibold text-amber-800">
                  {diagnosis.severity}
                </div>
              </div>

              <div className="space-y-2">
                <div className="flex justify-between text-sm">
                  <span>Lawn Health Score</span>
                  <span className="font-bold">{diagnosis.score}/100</span>
                </div>
                <div className="h-3 rounded-full bg-slate-100">
                  <div className="h-3 rounded-full bg-emerald-600" style={{ width: `${diagnosis.score}%` }} />
                </div>
                <p className="text-xs text-slate-500">Confidence: {diagnosis.confidence}</p>
              </div>

              <p className="rounded-2xl bg-slate-50 p-4 text-sm leading-6 text-slate-700">{diagnosis.summary}</p>

              <div className="grid grid-cols-3 gap-2">
                <div className="rounded-2xl bg-blue-50 p-3 text-center">
                  <Droplets className="mx-auto h-5 w-5" />
                  <p className="mt-1 text-xs">Water</p>
                </div>
                <div className="rounded-2xl bg-yellow-50 p-3 text-center">
                  <Sun className="mx-auto h-5 w-5" />
                  <p className="mt-1 text-xs">Light</p>
                </div>
                <div className="rounded-2xl bg-red-50 p-3 text-center">
                  <AlertTriangle className="mx-auto h-5 w-5" />
                  <p className="mt-1 text-xs">Risk</p>
                </div>
              </div>

              <div>
                <h3 className="font-semibold mb-3">Recommended Fix Plan</h3>
                <div className="space-y-2">
                  {diagnosis.steps.map((step, index) => (
                    <div key={step} className="flex gap-3 rounded-2xl bg-white border border-slate-100 p-3">
                      <CheckCircle2 className="mt-0.5 h-5 w-5 text-emerald-700" />
                      <p className="text-sm"><span className="font-semibold">Step {index + 1}:</span> {step}</p>
                    </div>
                  ))}
                </div>
              </div>

              <Button className="w-full rounded-2xl bg-slate-900 py-6 text-base hover:bg-slate-800">
                Save Recovery Plan
              </Button>
            </CardContent>
          </Card>
        )}
      </div>
    </div>
  );
}
